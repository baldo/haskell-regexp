1 {-# LANGUAGE MultiParamTypeClasses, FlexibleContexts #-}
    2 
    3 module Text.RegExp.Matching where
    4 
    5 import Data.Semiring
    6 import Text.RegExp.Data
    7 
    8 -- |
    9 -- Checks whether a regular expression matches the given word. For
   10 -- example, @acceptFull (fromString \"b|abc\") \"b\"@ yields @True@
   11 -- because the first alternative of @b|abc@ matches the string
   12 -- @\"b\"@.
   13 -- 
   14 acceptFull :: RegExp c -> [c] -> Bool
   15 acceptFull r = fullMatch r
   16 
   17 -- |
   18 -- Checks whether a regular expression matches a subword of the given
   19 -- word. For example, @acceptPartial (fromString \"b\") \"abc\"@
   20 -- yields @True@ because @\"abc\"@ contains the substring @\"b\"@.
   21 -- 
   22 acceptPartial :: RegExp c -> [c] -> Bool
   23 acceptPartial r = partialMatch r
   24 
   25 -- |
   26 -- Computes in how many ways a word can be matched against a regular
   27 -- expression.
   28 -- 
   29 matchingCount :: Num a => RegExp c -> [c] -> a
   30 matchingCount r = getNumeric . fullMatch r
   31 
   32 {-# SPECIALIZE matchingCount :: RegExp c -> [c] -> Int #-}
   33 
   34 -- |
   35 -- Matches a regular expression against a word computing a weight in
   36 -- an arbitrary semiring.
   37 -- 
   38 -- The symbols can have associated weights associated by the
   39 -- 'symWeight' function of the 'Weight' class. This function also
   40 -- allows to adjust the type of the used alphabet such that, for
   41 -- example, positional information can be taken into account by
   42 -- 'zip'ping the word with positions.
   43 -- 
   44 fullMatch :: Weight a b w => RegExp a -> [b] -> w
   45 fullMatch (RegExp r) = matchW (weighted r)
   46 
   47 {-# SPECIALIZE fullMatch :: RegExp c -> [c] -> Bool #-}
   48 {-# SPECIALIZE fullMatch :: RegExp c -> [c] -> Numeric Int #-}
   49 {-# SPECIALIZE fullMatch :: Num a => RegExp c -> [c] -> Numeric a #-}
   50 
   51 -- |
   52 -- Matches a regular expression against substrings of a word computing
   53 -- a weight in an arbitrary semiring. Similar to 'fullMatch' the
   54 -- 'Weight' class is used to associate weights to the symbols of the
   55 -- regular expression.
   56 -- 
   57 partialMatch :: Weight a b w => RegExp a -> [b] -> w
   58 partialMatch (RegExp r) = matchW (arb `seqW` weighted r `seqW` arb)
   59  where RegExp arb = rep anySym
   60 
   61 {-# SPECIALIZE partialMatch :: RegExp c -> [c] -> Bool #-}
   62 {-# SPECIALIZE partialMatch :: RegExp c -> [c] -> Numeric Int #-}
   63 {-# SPECIALIZE partialMatch :: Num a => RegExp c -> [c] -> Numeric a #-}
   64 
   65 matchW :: Semiring w => RegW w c -> [c] -> w
   66 matchW r []     = empty r
   67 matchW r (c:cs) = final (foldl (shiftW zero) (shiftW one r c) cs)
   68 
   69 {-# SPECIALIZE matchW :: RegW Bool c -> [c] -> Bool #-}
   70 {-# SPECIALIZE matchW :: RegW (Numeric Int) c -> [c] -> Numeric Int #-}
   71 {-# SPECIALIZE matchW :: Num a => RegW (Numeric a) c -> [c] -> Numeric a #-}
   72 
   73 shiftW :: Semiring w => w -> RegW w c -> c -> RegW w c
   74 shiftW w r c | active r || w /= zero = shift w (reg r) c
   75              | otherwise             = r
   76 
   77 shift :: Semiring w => w -> Reg w c -> c -> RegW w c
   78 shift _ Eps       _ = epsW
   79 shift w (Sym s f) c = let w' = w .*. f c
   80                        in (symW s f) { active = w' /= zero, final_ = w' }
   81 shift w (Alt p q) c = altW (shiftW w p c) (shiftW w q c)
   82 shift w (Seq p q) c = seqW (shiftW w p c)
   83                            (shiftW (w .*. empty p .+. final p) q c)
   84 shift w (Rep r)   c = repW (shiftW (w .+. final r) r c)
