1 {-# LANGUAGE MultiParamTypeClasses, FlexibleContexts #-}
    2 
    3 module Text.RegExp.Matching where
    4 
    5 import Data.Semiring
    6 import Text.RegExp.Data
    7 
    8 -- |
    9 -- Checks whether a regular expression matches the given word. For
   10 -- example, @accept (fromString \"b|abc\") \"b\"@ yields @True@
   11 -- because the first alternative of @b|abc@ matches the string
   12 -- @\"b\"@.
   13 -- 
   14 accept :: RegExp c -> [c] -> Bool
   15 accept r = fullMatch r
   16 
   17 -- |
   18 -- Computes in how many ways a word can be matched against a regular
   19 -- expression.
   20 -- 
   21 matchingCount :: Num a => RegExp c -> [c] -> a
   22 matchingCount r = getNumeric . fullMatch r
   23 
   24 -- |
   25 -- Matches a regular expression against a word computing a weight in
   26 -- an arbitrary semiring.
   27 -- 
   28 fullMatch :: Semiring w => RegExp c -> [c] -> w
   29 fullMatch (RegExp r) = matchW r
   30 
   31 -- |
   32 -- Matches a regular expression against substrings of a word computing
   33 -- a weight in an arbitrary semiring. The 'symWeight' function of
   34 -- 'Weight's is used to report positional information about the
   35 -- matching part of the word to the semiring.
   36 -- 
   37 partialMatch :: Weight c (Int,c) w => RegExp c -> [c] -> w
   38 partialMatch (RegExp r) =
   39   matchW (arb `seqW` weighted r `seqW` arb) . zip [(0::Int)..]
   40  where arb = repW (symW "." (const one))
   41 
   42 matchW :: Semiring w => RegW w c -> [c] -> w
   43 matchW r []     = empty r
   44 matchW r (c:cs) = final (foldl (shiftW zero) (shiftW one r c) cs)
   45 
   46 shiftW :: Semiring w => w -> RegW w c -> c -> RegW w c
   47 shiftW w r c | active r || w /= zero = shift w (reg r) c
   48              | otherwise             = r
   49 
   50 shift :: Semiring w => w -> Reg w c -> c -> RegW w c
   51 shift _ Eps       _ = epsW
   52 shift w (Sym s f) c = let w' = w .*. f c
   53                        in (symW s f) { active = w' /= zero, final_ = w' }
   54 shift w (Alt p q) c = altW (shiftW w p c) (shiftW w q c)
   55 shift w (Seq p q) c = seqW (shiftW w p c)
   56                            (shiftW (w .*. empty p .+. final p) q c)
   57 shift w (Rep r)   c = repW (shiftW (w .+. final r) r c)
