   22 -- |
   23 -- A semiring is an additive commutative monoid with identity 'zero':
   24 -- 
   25 -- >         a .+. b  ==  b .+. a
   26 -- >      zero .+. a  ==  a
   27 -- > (a .+. b) .+. c  ==  a .+. (b .+. c)
   28 -- 
   29 -- A semiring is a multiplicative monoid with identity 'one':
   30 -- 
   31 -- >        one .*. a  ==  a
   32 -- >        a .*. one  ==  a
   33 -- >  (a .*. b) .*. c  ==  a .*. (b .*. c)
   34 -- 
   35 -- Multiplication distributes over addition:
   36 -- 
   37 -- > a .*. (b .+. c)  ==  (a .*. b) .+. (a .*. c)
   38 -- > (a .+. b) .*. c  ==  (a .*. c) .+. (b .*. c)
   39 -- 
   40 -- 'zero' annihilates a semiring with respect to multiplication:
   41 -- 
   42 -- > zero .*. a  ==  zero
   43 -- > a .*. zero  ==  zero
   44 -- 
   45 -- All laws should hold with respect to the required `Eq` instance.
   46 -- 
   47 -- For example, the Booleans form a semiring.
   48 -- 
   49 --  * @False@ is an identity of disjunction which is commutative and
   50 --    associative,
   51 -- 
   52 --  * @True@ is an identity of conjunction which is associative,
   53 -- 
   54 --  * conjunction distributes over disjunction, and
   55 -- 
   56 --  * @False@ annihilates the Booleans with respect to conjunction.
   57 -- 
   58 class Eq s => Semiring s where
   59   zero, one    :: s
   60   (.+.), (.*.) :: s -> s -> s
   61 
   62 -- | Auxiliary function to convert Booleans to an arbitrary semiring.
   63 -- 
   64 fromBool :: Semiring s => Bool -> s
   65 fromBool False = zero
   66 fromBool True  = one
   67 
   68 instance Semiring Bool where
   69   zero = False; one = True; (.+.) = (||); (.*.) = (&&)
   70 
   71 -- |
   72 -- Wrapper for numeric types.
   73 -- 
   74 -- Every numeric type that satisfies the semiring laws (as all
   75 -- predefined numeric types do) is a semiring.
   76 -- 
   77 data Numeric a = Numeric { getNumeric :: a }
   78  deriving Eq
   79 
   80 instance (Num a, Show a) => Show (Numeric a) where
   81   show = show . getNumeric
   82 
   83 lift :: Num a => (a -> a) -> Numeric a -> Numeric a
   84 lift f = Numeric . f . getNumeric
   85 
   86 lift2 :: Num a => (a -> a -> a) -> Numeric a -> Numeric a -> Numeric a
   87 lift2 f x y = Numeric (f (getNumeric x) (getNumeric y))
   88 
   89 instance Num a => Num (Numeric a) where
   90   fromInteger = Numeric . fromInteger
   91   signum      = lift signum
   92   abs         = lift abs
   93 
   94   0 + x = x
   95   x + 0 = x
   96   x + y = lift2 (+) x y
   97 
   98   0 * _ = 0
   99   _ * 0 = 0
  100   1 * x = x
  101   x * 1 = x
  102   x * y = lift2 (*) x y
  103 
  104 instance Num a => Semiring (Numeric a) where
  105   zero = 0; one = 1; (.+.) = (+); (.*.) = (*)
