   21 
   22 import Text.RegExp
   23 
   24 -- |
   25 -- Subwords of words that match a regular expression are represented
   26 -- as values of type 'Matching'.
   27 -- 
   28 data Matching = Matching {
   29  
   30   -- | Start index of the matching subword in the queried word.
   31   matchingIndex :: !Int,
   32  
   33   -- | Length of the matching subword.
   34   matchingLength :: !Int
   35  
   36   }
   37  deriving Eq
   38 
   39 instance Show Matching
   40  where
   41   showsPrec _ m = showString "<index:" . shows (matchingIndex m)
   42                 . showString " length:" . shows (matchingLength m)
   43                 . showString ">"
   44 
   45 -- |
   46 -- Returns the leftmost longest of all non-empty matchings for a
   47 -- regular expression in a given word. If the empty word is the only
   48 -- matching its position is zero.
   49 -- 
   50 matching :: RegExp c -> [c] -> Maybe Matching
   51 matching r = getLeftLong . partialMatch r
   52 
   53 -- | 
   54 -- Semiring used for leftmost longest matching.
   55 -- 
   56 -- The `LeftLong` type satisfies the distributive laws only with a
   57 -- precondition on all involved multiplications: multiplied matches
   58 -- must be adjacent and the start position must be smaller than the
   59 -- end position. This precondition is satisfied for all
   60 -- multiplications during regular expression matching.
   61 -- 
   62 data LeftLong = Zero | One | LeftLong !Int !Int
   63  deriving (Eq,Show)
   64 
   65 getLeftLong :: LeftLong -> Maybe Matching
   66 getLeftLong Zero            =  Nothing
   67 getLeftLong One             =  Just $ Matching 0 0
   68 getLeftLong (LeftLong x y)  =  Just $ Matching x (y-x+1)
   69 
   70 instance Semiring LeftLong where
   71   zero = Zero; one = One
   72 
   73   Zero          .+.  y             =  y
   74   x             .+.  Zero          =  x
   75   One           .+.  y             =  y
   76   x             .+.  One           =  x
   77   LeftLong a b  .+.  LeftLong c d
   78     | a<c || a==c && b>=d          =  LeftLong a b
   79     | otherwise                    =  LeftLong c d
   80 
   81   Zero          .*.  _             =  Zero
   82   _             .*.  Zero          =  Zero
   83   One           .*.  y             =  y
   84   x             .*.  One           =  x
   85   LeftLong a _  .*.  LeftLong _ b  =  LeftLong a b
   86 
   87 instance Weight c (Int,c) LeftLong where
   88   symWeight p (n,c) = p c .*. LeftLong n n
