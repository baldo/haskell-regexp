   10 -- This module implements leftmost matching based on weighted regular
   11 -- expressions. It should be imported qualified as the interface
   12 -- resembles that provided by other matching modules.

   14 module Text.RegExp.Matching.Leftmost (
   15 
   16   Leftmost(..), Matching(..),
   17 
   18   matching, getLeftmost
   19 
   20   ) where
   21 
   22 import Text.RegExp
   23 
   24 -- |
   25 -- A 'Matching' records the leftmost start index of a matching subword.
   26 -- 
   27 data Matching = Matching {
   28  
   29   -- | Start index of the matching subword in the queried word.
   30   matchingIndex :: !Int
   31  
   32   }
   33  deriving Eq
   34 
   35 instance Show Matching
   36  where
   37   showsPrec _ m = showString "<index:" . shows (matchingIndex m)
   38                 . showString ">"
   39 
   40 -- |
   41 -- Returns the leftmost of all non-empty matchings for a regular
   42 -- expression in a given word. If the empty word is the only matching
   43 -- its position is zero.
   44 -- 
   45 matching :: RegExp c -> [c] -> Maybe Matching
   46 matching r = getLeftmost . partialMatch r
   47 
   48 -- | Semiring used for leftmost matching.
   49 -- 
   50 data Leftmost = Zero | One | Leftmost !Int
   51  deriving (Eq,Show)
   52 
   53 getLeftmost :: Leftmost -> Maybe Matching
   54 getLeftmost Zero          =  Nothing
   55 getLeftmost One           =  Just $ Matching 0 
   56 getLeftmost (Leftmost x)  =  Just $ Matching x
   57 
   58 instance Semiring Leftmost where
   59   zero = Zero; one = One
   60 
   61   Zero        .+.  y           =  y
   62   x           .+.  Zero        =  x
   63   One         .+.  y           =  y
   64   x           .+.  One         =  x
   65   Leftmost a  .+.  Leftmost b  =  Leftmost (min a b)
   66 
   67   Zero        .*.  _           =  Zero
   68   _           .*.  Zero        =  Zero
   69   One         .*.  y           =  y
   70   x           .*.  One         =  x
   71   Leftmost a  .*.  Leftmost b  =  Leftmost (min a b)
   72 
   73 instance Weight c (Int,c) Leftmost where
   74   symWeight p (n,c) = p c .*. Leftmost n
